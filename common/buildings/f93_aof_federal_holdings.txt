building_aof_federal_holdings = {
	base_buildtime = @b2_time
	# owner_type = subject_holding
	icon = building_aof_federal_holdings
	planet_limit = 1

	category = trade
	potential = {
		# always = yes
		is_sector_capital = yes
		owner = {
			exists = federation
			federation = {
				has_federation_law = succession_type_election
			}
		}
	}
	building_sets = {
		government
		urban
	}
	allow = {
		custom_tooltip = {
			fail_text = aof_need_permission_to_build
			owner = { check_variable = { which = aof_fed_permit value > 0 } }
		}
	}
	on_queued = {
		owner = { add_aof_fed_permit = { ADD = -1 } }
		# this has access to the variables "local_spent_<resource>"
	}
	
	on_unqueued = {
		owner = { add_aof_fed_permit = { ADD = 1 } }
		# this has access to the variables "local_spent_<resource>"
	}
	destroy_trigger = {
		is_sector_capital = no
		owner = {
			NOT = { exists = federation }
			AND = {
				exists = federation
				federation = { NOT = { has_federation_law = succession_type_election } }
			}
		}
	}
	on_destroy = {
		owner = {
			if = {
				limit = {
					exists = federation
					federation = {
						has_federation_law = succession_type_election
					}
				}
				add_aof_fed_permit = { ADD = 1 }
			}
			else = {
				add_resource = { trade = 15000 }
			}
		}
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				exists = federation
				federation = { federation_level > 0 }
			}
		}
		job_trader_add = 50
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				exists = federation
				federation = { federation_level > 1 }
			}
		}
		job_trader_add = 100
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				exists = federation
				federation = { federation_level > 2 }
			}
		}
		job_trader_add = 150
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				exists = federation
				federation = { federation_level > 3 }
			}
		}
		job_trader_add = 200
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				exists = federation
				federation = { federation_level > 4 }
			}
		}
		job_trader_add = 250
	}
	triggered_planet_modifier = {
		potential = {
			owner = {
				exists = federation
				federation = { federation_level > 5 } # only with another mod increasing fed levels
			}
		}
		job_trader_add = 300
	}

	resources = {
		category = planet_holdings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = @b1_upkeep
			consumer_goods = 5
		}
	}
	ai_weight_coefficient = 1000
}
