fed_accords_coop = {
	potential = {}
	allow = {
		custom_tooltip = {
			NOT = {
				AND = {
					NOT = { has_federation_law = fed_accords_coop }

					has_any_federation_law_in_category = law_category_aof_accords

			}

		has_federation_law = assembly_grand
		OR = {
			has_federation_law = treaties_separate_no

		}
	}
	modifier = {
		federation_research_sharing_mult = 0.1
		federation_experience_add = 4
	}
	required_centralization = {
		base = 4
	}
	cohesion_growth = -2
	on_enact = {
		every_member = {
			remove_modifier = f93_aof_accord_unyielding
			remove_country_flag = aof_agreed_to_fed_accords_unyielding
			add_modifier = { modifier = f93_aof_accord_coop mult = 1 }
		}
		add_cohesion = @cohesion_loss
		if = {
			limit = { has_federation_flag = enable_federation_cooldowns }
			add_modifier = {
				modifier = federation_law_category_aof_accords_cooldown
				days = @cooldown
			}
		}
	}
	ai_weight = { 
		base = 0
		
		#Default support
		modifier = {
			desc = federation_acceptance_generally_positive
			add = 20
		}
		modifier = {
			desc = federation_acceptance_materialist
			add = 15
			is_materialist = yes
		}
		modifier = {
			desc = federation_acceptance_spiritualist
			add = -10
			is_spiritualist = yes
		}
		modifier = {
			desc = federation_acceptance_militarist
			add = 5
			is_militarist = yes
		}
		modifier = {
			desc = federation_acceptance_weaker
			add = 25
			relative_power = { who = from category = technology value >= superior }
		}
		modifier = {
			desc = federation_acceptance_erudite_explorers
			add = 14
			has_ai_personality = erudite_explorers
		}
		modifier = {
			desc = federation_acceptance_technocracy
			add = 30

		}
	}
