@cooldown = 3600
@cohesion_loss = -20

diplomatic_bloc_default = {
	potential = {}
	allow = {
		has_federation_law = vote_weight_equal
		custom_tooltip = {
			NOT = {
				AND = {
					NOT = { has_federation_law = diplomatic_bloc_default }
					has_modifier = federation_law_category_diplomatic_block_cooldown
					has_any_federation_law_in_category = law_category_diplomatic_block
				}
			}
		}
	}
	modifier = {
		federation_leader_diplo_weight_contribution_add = 0.1
	}
	on_enact = {
		if = {
			limit = { has_federation_flag = enable_federation_cooldowns }
			add_modifier = {
				modifier = federation_law_category_diplomatic_block_cooldown
				days = @cooldown
			}
		}
	}
	ai_weight = { 
		base = 0
		
		#Default support
		modifier = {
			desc = federation_acceptance_generally_positive
			add = 41
		}
	}
}
