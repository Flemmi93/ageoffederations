@cooldown = 3600
@cohesion_loss = -20

fed_market_none = {
	potential = {
		NOR = {
			has_federation_law = fed_market_basic
			has_federation_law = fed_market_unity
			has_federation_law = fed_market_military
			has_federation_law = fed_market_strategic
		}
	}
}
fed_market_basic = {
	potential = {}
	allow = {
		custom_tooltip = {
			NOT = {
				AND = {
					NOT = { has_federation_law = fed_market_basic }
					has_modifier = federation_law_category_aof_market_cooldown
					has_any_federation_law_in_category = law_category_aof_market
				}
			}
		}
		custom_tooltip = {
			has_federation_flag = federation_one_enabled
			fail_text = no_federation_one
		}
	}
	required_centralization = {
		base = 2
	}
	modifier = {}
	on_enact = {
		add_cohesion = @cohesion_loss
		if = {
			limit = { has_federation_flag = enable_federation_cooldowns }
			add_modifier = {
				modifier = federation_law_category_aof_market_cooldown
				days = @cooldown
			}
		}
		every_member = {
			remove_market_modifier = yes
			add_modifier = {
				modifier = f93_aof_market_basic
				mult = value:f93_market_policy_basic
			}
			country_event = { id = f93_aof_market.1 }
		}
	}
	ai_weight = { 
		base = 0
		
		#Default support
		modifier = {
			desc = federation_acceptance_generally_positive
			add = 30
		}
		modifier = {
			desc = f93_has_federation_one
			add = 50
			any_owned_planet = {
				has_planet_flag = f93_federation_one
			}
		}
	}
}
