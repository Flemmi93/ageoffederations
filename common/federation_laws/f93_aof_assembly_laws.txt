assembly_high = {
	potential = {}
	allow = {
		custom_tooltip = {
			NOT = {
				AND = {
					NOT = { has_federation_law = assembly_high }
					has_modifier = federation_law_category_assembly_cooldown
					has_any_federation_law_in_category = law_category_assembly
				}
			}
		}
	}
	modifier = {}
	on_enact = {
		add_cohesion = @cohesion_loss
		if = {
			limit = { has_federation_flag = enable_federation_cooldowns }
			add_modifier = {
				modifier = federation_law_category_assembly_cooldown
				days = @cooldown
			}
		}
		if = {
			limit = {
				has_federation_flag = had_grand_assembly
				NOT= { has_federation_flag = already_downgraded_assembly_before }
			}
			every_member = { country_event = { id = f93_aof_background.10 } }
			set_federation_flag = already_downgraded_assembly_before
		}
	}
	ai_weight = { 
		base = 0
		
		#Default support
		modifier = {
			desc = federation_acceptance_generally_positive
			add = 41
		}
		modifier = {
			desc = aof_wont_go_back_cant_go_back
			add = -50
			federation = { has_federation_law = assembly_grand }
		}
	}
}
