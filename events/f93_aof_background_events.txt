country_event = {
	id = f93_aof_background.1
	is_triggered_only = yes
	title = f93_aof_background.1.name
	desc = f93_aof_background.1.desc
	picture = GFX_evt_hand_shake
	immediate = {
		every_country = {
			limit = {
				has_communications = root
				NOT = { is_in_federation_with = root }
				NOT = { has_association_status = root }
			}
			end_all_treaties_with = root
			root = {
				end_all_treaties_with = prev
			}
		}
	}
	option = {
		name = f93_aof_background.1.a
		if = {
			limit = { has_policy_flag = slavery_allowed }
			set_policy = { policy = slavery option = slavery_not_allowed cooldown = yes }
			set_country_flag = aof_agreed_to_treaties_separate_ethical_partners
		}
		if = {
			limit = { has_policy_flag = interference_aggressive }
			set_policy = { policy = interference option = interference_active cooldown = yes }
			set_country_flag = aof_agreed_to_treaties_separate_ethical_partners
		}
		if = {
			limit = { has_policy_flag = refugees_not_allowed }
			set_policy = { policy = refugees option = refugees_only_citizens cooldown = yes }
			set_country_flag = aof_agreed_to_treaties_separate_ethical_partners
		}
		ai_chance = {
			base = 10
		}
	}
	option = {
		name = f93_aof_background.1.b
		leave_alliance = {
			override_requirements = yes
			apply_opinion_penalty = yes
		}
		ai_chance = {
			base = 0
			modifier = {
				has_policy_flag = slavery_allowed
				add = 15
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				has_origin = origin_syncretic_evolution
				add = 15
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				has_policy_flag = interference_aggressive
				add = 5
			}
		}
	}
}

# no confidence counter
country_event = {
	id = f93_aof_background.4
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		exists = federation
		is_federation_leader = yes
		any_federation = {
			any_member = { is_same_value = root }
			federation_cohesion <= 0
			has_federation_flag = first_election_happened
			NOT = { has_federation_flag = ongoing_election_process }
		}
		NOT = { has_country_flag = no_confidence_happened }
	}
	immediate = {
		if = {
			limit = { NOT = { is_variable_set = no_confidence_counter } }
			set_variable = { which = no_confidence_counter value = 1 }
		}
		set_timed_country_flag = { flag = no_confidence_counter days = 400 }
		change_variable = { which = no_confidence_counter value = 1 }
		if = {
			limit = { check_variable = { which = no_confidence_counter value = 12 } }
			set_timed_country_flag = { flag = no_confidence_happened days = 400 } # to make sure the counter can reset at least
			set_variable = { which = no_confidence_counter value = 0 }
			country_event = { id = f93_aof_background.5 days = 5 random = 10 }
		}
	}
}

# THIS = System: system being occupied
# FROM = Country: Conqueror of the system
# FROMFROM = Country: Original owner of the system
system_event = {
	id = f93_aof_background.13
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		any_ambient_object = {
			has_ambient_object_flag = f93_federation_one_station
		}
	}
	immediate = {
		FROMFROM = { federation = { every_member = { country_event = { id = f93_aof_background.131 } } } }
		every_system_ambient_object = {
			limit = { has_ambient_object_flag = f93_federation_one_station }
			destroy_ambient_object = this
		}
		every_system_planet = {
			remove_planet_flag = f93_has_federation_one
		}
		random_galaxy_planet = {
			limit = { owner = { is_in_federation_with = FROMFROM } }
			set_planet_flag = f93_has_federation_one
			save_event_target_as = f93_has_federation_one
			if = {
				limit = { has_megacorp = yes }
				create_ambient_object = { type = "f93_federation_one_station" location = this }
			}
			else = {
				create_ambient_object = { type = "f93_federation_one_station_no_dlc" location = this }
			}
			last_created_ambient_object = {
				set_ambient_object_flag = f93_federation_one_station
				set_location = {
					target = prev
					distance = 0
				}
			}
		}
	}
}
