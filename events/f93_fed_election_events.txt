		}

		name = f93_fed_election.1.nominate
		allow = {

			country_event = { id = f93_fed_election.1 }
		}
		ai_chance = {
			base = 30
			modifier = {
				factor = 2

					resource = influence

				}
			}
			modifier = {
				factor = 2
				resource_income_compare = {
					resource = influence
					value >= 2
				}
			}
		}

		allow = {
			has_country_flag = f93_fed_nominated
			NOT = { has_country_flag = f93_fed_nomination_boosted_2 }
		}
		hidden_effect = {
			remove_country_flag = election_ui_open
			country_event = { id = f93_fed_election.1 }

				limit = { has_country_flag = f93_fed_nomination_boosted_1 }
				set_country_flag = f93_fed_nomination_boosted_2
			}
			else = { set_country_flag = f93_fed_nomination_boosted_1 }
		}
		
		ai_chance = {
			base = 30
			modifier = {
				factor = 2
				resource_income_compare = {
					resource = influence
					value >= 3
				}
			}
		}
	}
	option = { #second boost
		name = f93_fed_election.1.boost_nominate
		trigger = { has_country_flag = f93_fed_nomination_boosted_1 }
		allow = {
			has_country_flag = f93_fed_nominated
			NOT = { has_country_flag = f93_fed_nomination_boosted_2 }
		}
		hidden_effect = {
			remove_country_flag = election_ui_open
			country_event = { id = f93_fed_election.1 }
			if = {
				limit = { has_country_flag = f93_fed_nomination_boosted_1 }
				set_country_flag = f93_fed_nomination_boosted_2
			}
			else = { set_country_flag = f93_fed_nomination_boosted_1 }
		}
		
		ai_chance = {
			base = 30
			modifier = {
				factor = 2
				resource_income_compare = {
					resource = influence
					value >= 4
				}
			}
		}
	}
	option = {
		name = f93_fed_election.1.revoke_nominate
		allow = {
			has_country_flag = f93_fed_nominated
		}

			remove_country_flag = election_ui_open
			country_event = { id = f93_fed_election.1 }
			remove_country_flag = f93_fed_nominated

country_event = {
	id = f93_fed_election.5
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		set_variable = { which = vote_result value = 0 }
		federation = {
			every_member = {
				limit = { has_country_flag = voted_to_keep }
				root = { change_variable = { which = vote_result value = 1 } }
			}
			every_member = {
				limit = { has_country_flag = voted_to_kick }
				root = { change_variable = { which = vote_result value = -1 } }
			}
		}
		if = {
			limit = { has_country_flag = voted_to_keep }
			change_variable = { which = vote_result value = 1 }
		}
		if = {
			limit = { has_country_flag = voted_to_kick }
			change_variable = { which = vote_result value = -1 }
		}
		if = {
			limit = { check_variable = { which = vote_result value < 0 } }
			country_event = { id = f93_fed_election.7 }
			federation = {
				federation_leader = {
					add_modifier = {
						modifier = f93_cohesion
						mult = -5
						days = 1800
					}
				}
				every_member = {
					if = {
						limit = { NOT = { has_country_flag = no_confidence_happened } }
						country_event = { id = f93_fed_election.1 days = 60 }
					}
					set_timed_country_flag = { flag = no_confidence_happened days = 180 }
				}
			}
		}
		else = {
			country_event = { id = f93_fed_election.6 }
			federation = {
				every_member = {
					set_timed_country_flag = { flag = no_confidence_happened days = 1080 } # 3 year cool-down
				}
				federation_leader = {
					add_modifier = {
						modifier = f93_cohesion
						mult = -2
						days = 1080
					}
				}
			}
		}
	}
}
