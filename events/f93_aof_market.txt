namespace = f93_aof_market
# Initializing market
country_event = {
	id = f93_aof_market.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		set_variable = { which = aof_market_volume value = 0 }
		every_galaxy_planet = {
			limit = {
				exists = owner
				owner = { exists = federation }
				owner = {
					is_in_federation_with = root
					# is_subject = no
				}
			}
			root = {
				if = {
					limit = { NOT = { is_variable_set = aof_market_volume } }
					set_variable = { which = aof_market_volume value = 0 }
				}
				change_variable = { which = aof_market_volume value = prev.value:f93_planet_trade_income_compare }
				if = {
					limit = { is_hive_empire = no }
					change_variable = { which = aof_market_volume value = prev.value:f93_planet_cg_income_compare }
				}
				# if = { # replaced with 0.5 cohesion
				# 	limit = { exists = federation federation = { has_federation_law = treaties_separate_no } }
				# 	multiply_variable = { which = aof_market_volume value = 1.2 } #20% market volume boost
				# }
				if  = {
					limit = {
						prev = { has_building = building_aof_federal_holdings }
						# has_holding = {
						# 	holding = holding_aof_federal_holdings
						# 	owner = planet.owner.federation.federation_leader
						# }
					}
					multiply_variable = { which = aof_market_volume value = 1.05 } #5% market volume boost
				}
			}			
		}
		round_variable = aof_market_volume

		#update modifier
		if = {
			limit = { has_modifier = f93_aof_market_basic }
			remove_market_modifier = yes
			add_modifier = {
				modifier = f93_aof_market_basic
				mult = value:f93_market_policy_basic
			}
		}
		else_if = {
			limit = { has_modifier = f93_aof_market_unity }
			remove_market_modifier = yes
			add_modifier = {
				modifier = f93_aof_market_unity
				mult = value:f93_market_policy_unity
			}
		}
		else_if = {
			limit = { has_modifier = f93_aof_market_military }
			remove_market_modifier = yes
			add_modifier = {
				modifier = f93_aof_market_military
				mult = value:f93_market_policy_military
			}
		}
		else_if = {
			limit = { has_modifier = f93_aof_market_strategic }
			remove_market_modifier = yes
			add_modifier = {
				modifier = f93_aof_market_strategic
				mult = value:f93_market_policy_strategic
			}
		}
	}
}

country_event = {

country_event = {
	id = f93_aof_market.5
	hide_window = yes
	is_triggered_only = yes
	trigger = { NOT = { has_global_flag = f93_federation_one_recently } } #avoid spam
	immediate = {
		set_timed_global_flag = { flag = f93_federation_one_recently days = 1 }
		every_ambient_object = {
			limit = { has_ambient_object_flag = f93_federation_one_station }
			destroy_ambient_object = this

		}
		every_galaxy_planet = {
			limit = { has_planet_flag = f93_has_federation_one }
			if = {
				limit = { has_megacorp = yes }
				create_ambient_object = { type = "f93_federation_one_station" location = this }
			}
			else = {
				create_ambient_object = { type = "f93_federation_one_station_no_dlc" location = this }
			}
			last_created_ambient_object = {
				set_ambient_object_flag = f93_federation_one_station
				set_location = {
					target = prev
					distance = 0
				}
			}
		}
	}
}
