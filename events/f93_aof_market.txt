namespace = f93_aof_market
# Initializing market
country_event = {
	id = f93_aof_market.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		set_variable = { which = aof_market_volume value = 0 }
		every_galaxy_planet = {
			limit = {
				exists = owner
				owner = {
					is_in_federation_with = root
					# is_subject = no
				}
			}
			root = {
				if = {
					limit = { NOT = { is_variable_set = aof_market_volume } }
					set_variable = { which = aof_market_volume value = 0 }
				}
				change_variable = { which = aof_market_volume value = prev.value:f93_planet_trade_income_compare }
				if = {
					limit = { is_hive_empire = no }
					change_variable = { which = aof_market_volume value = prev.value:f93_planet_cg_income_compare }
				}
				if = {
					limit = { exists = federation federation = { has_federation_law = treaties_separate_no } }
					multiply_variable = { which = aof_market_volume value = 1.1 }
				}
			}			
		}
		if  = {
			limit = { has_modifier = f93_aof_market_extract }
			add_modifier = {
				modifier = f93_aof_market_extract
				mult = aof_market_volume
			}
		}
		if  = {
			limit = { has_modifier = f93_aof_market_mix }
			add_modifier = {
				modifier = f93_aof_market_mix
				mult = aof_market_volume
			}
		}
		if  = {
			limit = { has_modifier = f93_aof_market_civilian }
			add_modifier = {
				modifier = f93_aof_market_civilian
				mult = aof_market_volume
			}
		}
		if  = {
			limit = { has_modifier = f93_aof_market_military }
			add_modifier = {
				modifier = f93_aof_market_military
				mult = aof_market_volume
			}
		}
		if  = {
			limit = { has_modifier = f93_aof_market_science }
			add_modifier = {
				modifier = f93_aof_market_science
				mult = aof_market_volume
			}
		}
	}
}

	}

country_event = {
	id = f93_aof_market.5
	hide_window = yes
	is_triggered_only = yes
	trigger = { has_global_flag = f93_federation_one_recently } #avoid spam
	immediate = {
		set_timed_global_flag = { flag = f93_federation_one_recently days = 1 }
		every_ambient_object = {
			limit = { has_ambient_object_flag = f93_federation_one_station }
			destroy_ambient_object = this

		}
		every_galaxy_planet = {
			limit = { has_planet_flag = f93_has_federation_one }
			create_ambient_object = { type = "f93_federation_one_station" location = this }
			last_created_ambient_object = {
				set_ambient_object_flag = f93_federation_one_station
				set_location = {
					target = prev
					distance = 0
				}
			}
		}
	}
}
