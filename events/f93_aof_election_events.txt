#Detects if an election should be triggered
country_event = {
	id = f93_aof_election.0
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		exists = federation
		is_federation_leader = yes
		federation = { NOT = { has_federation_flag = ongoing_election_process } }
		OR = {
			federation = {
				has_federation_law = succession_type_election
				has_federation_flag = first_election_happened
				AND = {
					has_federation_flag = federation_recent_election # to percent false positives
					federation_leader = { NOT = { has_modifier = f93_aof_president_country } } 
				}
			}
			federation = {
				has_federation_law = succession_type_election
				NOT = { has_federation_flag = f93_aof_election }
				has_federation_flag = first_election_happened
			}
		}
	}
	immediate = {
		federation = {
			every_member = { country_event = { id = f93_aof_election.1 } }
			set_federation_flag = ongoing_election_process
		}
	}
}

country_event = {
	id = f93_aof_election.5
	is_triggered_only = yes
	hide_window = yes
	immediate = {
		set_variable = { which = vote_result value = 0 }
		federation = {
			every_member = {
				limit = { has_country_flag = voted_to_keep }
				root = { change_variable = { which = vote_result value = 1 } }
			}
			every_member = {
				limit = { has_country_flag = voted_to_kick }
				root = { change_variable = { which = vote_result value = -1 } }
			}
		}
		if = {
			limit = { has_country_flag = voted_to_keep }
			change_variable = { which = vote_result value = 1 }
		}
		if = {
			limit = { has_country_flag = voted_to_kick }
			change_variable = { which = vote_result value = -1 }
		}
		if = {
			limit = { check_variable = { which = vote_result value < 0 } }
			country_event = { id = f93_aof_election.7 }
			federation = {
				federation_leader = {
					add_modifier = {
						modifier = f93_cohesion
						mult = -5
						days = 1800
					}
				}
				every_member = {
					if = {
						limit = { NOT = { has_country_flag = no_confidence_happened } }
						country_event = { id = f93_aof_election.1 days = 60 }
					}
					set_timed_country_flag = { flag = no_confidence_happened days = 180 }
				}
			}
		}
		else = {
			country_event = { id = f93_aof_election.6 }
			federation = {
				every_member = {
					set_timed_country_flag = { flag = no_confidence_happened days = 1080 } # 3 year cool-down
				}
				federation_leader = {
					add_modifier = {
						modifier = f93_cohesion
						mult = -2
						days = 1080
					}
				}
			}
		}
	}
}
